class Worker:
    def __init__(self, x, y, z, alpha, lam, beta):
        self.x = x #local feature matrix is s-by-d
        self.y = y #local label matrix is s-by-1
        self.s = x.shape[0] #given no of local samples
        self.d = x.shape[1] #given no of features
        self.w = numpy.zeros((self.d, 1)) #model parameter vector is d-by-1 
        self.q = z #No of times local gradient to be performed
        self.v = numpy.zeros((self.d, 1)) #momentum is d-by1
        self.alpha = alpha #Local learning rate
        self.initial_w = numpy.zeros((self.d, 1)) #For finding direction at the end we need to store the w broadcasted by server
        self.lam = lam #regularization parameter
        self.beta = beta #momentum parameter 
    #Set the model parameters to the latest
    def set_param(self, w):
        self.w = w
        self.initial_w = w
    
    #compute the local loss
    def loss(self):
        yx = numpy.multiply(self.y, self.x) #s-by-d matrix
        yxw = numpy.dot(yx, self.w) #s-by-1 matrix
        vec1 = numpy.exp(-yxw) #s-by-1 matrix
        vec2 = numpy.log(1 + vec1) #s-by-1 matrix
        reg = self.lam / 2*numpy.sum(self.w * self.w)
        return numpy.sum(vec2)/self.s + reg #loss function
    
    #Compute the local gradient 
    def gradient(self):
        yx = numpy.multiply(self.y, self.x) # s-by-d matrix
        yxw = numpy.dot(yx, self.w) # s-by-1 matrix
        vec1 = numpy.exp(yxw) # s-by-1 matrix
        vec2 = numpy.divide(yx, 1+vec1) # s-by-d matrix
        reg = self.lam / 2*numpy.sum(self.w * self.w)
        g = -numpy.mean(vec2, axis=0).reshape(self.d, 1)+reg # d-by-1 matrix
        return g
 
    #Compute q gradient 
    def gradient_descent_local(self):
        for i in range(self.q):
            gradient = self.gradient()
            self.v = self.beta*self.v + gradient
            self.w -= self.alpha * self.v
            losses = self.loss()
            print("The local objective func_value is given by", losses)
        return self.w
    
    #Compute Stochastic gradient descent
    def sgd(self, i, x_rand, y_rand):
        import numpy as np
        xi = x_rand[i,:]
        yi = float(y_rand[i,:])
        # calculate the stochastic gradient and local worker loss
        d = xi.shape[0]
        yx = yi * xi # 1-by-d matrix
        yxw = float(np.dot(yx, self.w)) #scalar
        loss = np.log(1 + np.exp(-yxw)) #scalar
        reg = lam / 2 * np.sum(self.w * self.w)#scalar
        obj = loss + reg
        g_loss = -yx.T / (1 + np.exp(yxw))# d-by-1 matrix
        g = g_loss + lam * self.w # d-by-1 matrix
        return g, obj
    
    #Compute q stochastic gradients 
    def sgd_local(self):
        import numpy as np
        for i in range(self.q):
            rand_indices = np.random.permutation(self.s)
            x_rand = self.x[rand_indices,:]
            y_rand = self.y[rand_indices,:]
            objval_stc = 0
            for i in range(self.s):
                g, obj = self.sgd(i, x_rand, y_rand)
                self.w -= self.alpha * g
                objval_stc+=obj
            self.alpha *= 0.9 # decrease step size
            objval_stc /= self.s
            print("local objective function value ", objval_stc)
        return self.w
    
    #Tracking direction such that sent back to server
    def track_direction(self):
        pk = self.initial_w - self.gradient_descent_local()
        return pk
    
    #Find direction to be sent back to server stochastic gradient 
    def track_stochdirection(self):
        pk = self.initial_w - self.sgd_local()
        return pk

class Server:
    def __init__(self, m, n, d, alpha, lam):
        self.m = m #number of worker nodes
        self.n = n #number of training samples
        self.d = d #number of features
        self.w = numpy.zeros((d, 1)) #d-by-1 model parameter vector
        self.pk = numpy.zeros((d, 1)) #directions recieved from different workers
        self.alpha = alpha #Server Learning rate
        self.lam = lam #Server reg parameter
        #self.g = numpy.zeros((d, 1)) #d-by-1 gradient
        #self.v = numpy.zeros((d, 1)) #d-by1 momentum
        self.loss = 0 #loss function value
        #self.obj = 0 #objective function value
    
    def broadcast(self):
        return self.w
    
    def aggregate(self, pk, losses):
        self.pk = numpy.mean(numpy.array(pk), axis=0)
        self.loss = sum(losses)
        
    def update_fun(self):
        self.w = self.w - self.alpha*self.pk
        self.alpha *= 0.9 # decrease step size
            
    #Compute the objective function 
    def objective(self):
        reg = self.lam/2 * numpy.sum(self.w*self.w)
        self.obj = self.loss/self.m + reg
        return self.obj

def create_server_workers(m, x, y, q, alpha, lam, beta):
    n, d = x.shape
